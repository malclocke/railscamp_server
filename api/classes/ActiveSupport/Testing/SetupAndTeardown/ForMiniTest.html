def run(runner)
  result = '.'
  begin
    run_callbacks :setup do
      result = super
    end
  rescue Exception => e
    result = runner.puke(self.class, method_name, e)
  ensure
    begin
      run_callbacks :teardown
    rescue Exception => e
      result = runner.puke(self.class, method_name, e)
    end
  end
  result
end
